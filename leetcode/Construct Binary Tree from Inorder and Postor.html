Construct Binary Tree from Inorder and Postorder Traversal

创建时间：2014/6/19 22:27
更新时间：2014/6/19 22:28

Given inorder and postorder traversal of a tree, construct the binary tree.

Note:
You may assume that duplicates do not exist in the tree.


/**
 * Definition for binary tree
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
    int[] inorder,postorder ;
    public TreeNode buildTree (int[] inorder, int[] postorder ) {
         if ( inorder == null || postorder == null
                 || inorder.length != postorder .length || inorder.length < 1 )
             return null ;
         this.inorder = inorder;
         this.postorder = postorder;
         return buildTree(0, inorder .length - 1, 0 ,
                postorder .length - 1);
    }

    public TreeNode buildTree (int is, int ie,
             int ps, int pe) {
         if ( ie < is )
             return null ;
         if ( ie == is )
             return new TreeNode(postorder [pe]);
        TreeNode root = new TreeNode(postorder [pe]);
         int rootPosition = ie;
         for(;rootPosition >= is; rootPosition-- ){
             if(inorder [rootPosition] == postorder [pe])
                 break;
         }
         int leftCount = rootPosition - is ;
        root .left = buildTree( is , rootPosition - 1, ps, ps
                 + leftCount - 1);
        root .right = buildTree(rootPosition + 1 , ie,
                ps + leftCount, pe - 1);
         return root;
    }
}
