Construct Binary Tree from Preorder and Inorder Traversal

创建时间：2014/6/20 13:47
更新时间：2014/6/20 13:47

Given preorder and inorder traversal of a tree, construct the binary tree.

Note:
You may assume that duplicates do not exist in the tree.

/**
 * Definition for binary tree
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
    int[] preorder, inorder ;

    public TreeNode buildTree (int[] preorder, int[] inorder ) {
         if ( preorder == null || inorder == null || preorder.length == 0
                 || inorder.length == 0)
             return null ;
         this.preorder = preorder;
         this.inorder = inorder;
         return buildTree(0, preorder .length - 1, 0 , inorder.length - 1);
    }

    public TreeNode buildTree (int preS, int preE, int inS, int inE) {
         if ( preS > preE )
             return null ;
        TreeNode root = new TreeNode(preorder [preS]);
         if ( preS == preE )
             return root;
         int posi;
         for ( posi = inS ; posi <= inE; posi ++) {
             if ( inorder[posi ] == preorder[preS ])
                 break;
         }
        root .left = buildTree(preS + 1 , preS + posi - inS , inS, posi - 1);
        root .right = buildTree(preS + posi - inS + 1, preE , posi + 1 , inE);
         return root;
    }
}
