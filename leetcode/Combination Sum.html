Combination Sum

创建时间：2014/6/19 21:37
更新时间：2014/6/19 21:39

Given a set of candidate numbers (C) and a target number (T), find all unique combinations in C where the candidate numbers sums to T.

The same repeated number may be chosen from C unlimited number of times.

Note:

  • All numbers (including target) will be positive integers.
  • Elements in a combination (a1, a2, … , ak) must be in non-descending order. (ie, a1 ≤ a2 ≤ … ≤ ak).
  • The solution set must not contain duplicate combinations.

For example, given candidate set 2,3,6,7 and target 7, 
A solution set is: 
[7] 
[2, 2, 3] 


public class Solution {
    List<List <Integer>> result;
    int[] candidates;
    int target ;

    public List <List< Integer>> combinationSum (int[] candidates, int target) {
        result = new ArrayList<List <Integer>>();
         if ( candidates == null || candidates.length == 0 )
             return result;
         if ( target == 0) {
            result .add( new ArrayList<Integer>());
             return result;
         }
        quickSort (candidates, 0, candidates.length - 1 );
         this.candidates = candidates;
         this.target = target;
        combinationSum (new ArrayList <Integer>(), 0, 0 );
         return result;
    }

    public void combinationSum (List< Integer> input , int sum, int begin) {
         if ( sum > target )
             return;
         else if (sum == target ) {
            result .add(( List<Integer >) (( ArrayList<Integer >) input).clone());
             return;
         } else {
             for ( int i = begin; i < candidates.length; i ++) {
                input .add( candidates[i ]);
                combinationSum (input, sum+candidates [i], i);
                input .remove( input.size ()-1);
             }
         }
    }

    public void quickSort (int[] num, int s, int e) {
         if ( s >= e)
             return;
         int p = partition(num , s, e);
        quickSort (num, s, p - 1);
        quickSort (num, p + 1, e );
    }

    public int partition (int[] num, int p, int q) {
         int posi = (int) (Math. random() * ( q - p) + p);
         int pivot = num[posi ];
        num [posi] = num[p ];
         while ( p < q) {
             while ( p < q && num [q] >= pivot)
                q --;
            num [p] = num[q ];
             while ( p < q && num [p] <= pivot)
                p ++;
            num [q--] = num[p ];
         }
        num [p] = pivot;
         return p;
    }
}
