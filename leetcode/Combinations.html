Combinations

创建时间：2014/6/15 16:15
更新时间：2014/6/15 16:15

Given two integers n and k, return all possible combinations of k numbers out of 1 ... n.

For example,
If n = 4 and k = 2, a solution is:

[
  [2,4],
  [3,4],
  [2,3],
  [1,2],
  [1,3],
  [1,4],
]
public class Solution {
    List<List <Integer>> result;

    public List <List< Integer>> combine (int n, int k) {
        result = new ArrayList<List <Integer>>();
         if ( k == 0 || n == 0 )
             return result;
        combine (n, k, new ArrayList <Integer>(), 0);
         return result;
    }

    public void combine (int n, int k, List<Integer > input, int start) {
         if ( k == 0) {
            result .add(( List<Integer >) ((ArrayList<Integer>)input ).clone());
             return;
         }
         for ( int i = start; i < n; i++) {
            input .add( i + 1);
            combine (n, k - 1, input , i + 1 );
            input .remove( input.size () - 1 );
         }
    }
}
