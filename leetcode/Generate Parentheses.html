Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses.

For example, given n = 3, a solution set is:

"((()))", "(()())", "(())()", "()(())", "()()()"

public class Solution {
   public List <String> generateParenthesis(int n) {
        List <String> result = new ArrayList<String>();
         int[] combine = new int [2 * n ];// 1 for } 0 for {
         int index = 0;

         while ( index >= 0 && index < 2 * n ) {
             if ( combine[index ] > 1 ) {
                combine [index--] = 0;
                combine [index]++;
             } else if (check (combine, n, index )) {
                 if ( index == 2 * n - 1 ) {
                    addResult (combine, result);
                    combine [index]++;
                 } else {
                    index ++;
                 }
             } else {
                 if(index == 0 && combine[0 ] != 0 ) break ;
                 if ( combine[index ] == 1 ) {
                    combine [index--] = 0;
                    combine [index]++;
                 } else {
                    combine [index]++;
                 }
             }
         }

         return result;
    }
    
    public void addResult (int[] combine,List <String> result){
        StringBuffer sb = new StringBuffer();
         for(int i=0;i <combine.length;i ++){
            sb .append( combine[i ]==0? "(":")" );
         }
        result .add( sb.toString());
    }

    public boolean check (int[] combine, int n, int index) {
         int lc, rc, posi ;
        posi = lc = rc = 0;
         for ( int i = 0 ; i <= index; i ++) {
             if(combine [i] == 0){
                lc ++;
                posi ++;
             }else{
                rc ++;
                posi --;
             }
             if(posi < 0 ) return false;
         }
         if(lc <= n && rc <= n )
             return true ;
         return false ;
    }
}
