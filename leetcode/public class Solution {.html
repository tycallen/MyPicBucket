Write a program to solve a Sudoku puzzle by filling the empty cells.

Empty cells are indicated by the character '.'.

You may assume that there will be only one unique solution.


A sudoku puzzle...


...and its solution numbers marked in red.

public class Solution {
    char[] rows, cols ;
    char[][] cubes;
    static final int MATRIX_SIZE = 9 ;
    char[][] board;
    Map<Integer , Integer> setRecord;

    public void solveSudoku (char[][] board) {
         if ( board == null || board.length != 9 || board[0 ].length != 9)
             return;
         this.board = board;
         int[] miss = new int [MATRIX_SIZE * MATRIX_SIZE];
         int missI = 0, missL;
        rows = new char [9];
        cols = new char [9];
        cubes = new char [3][ 3];
        setRecord = new HashMap<Integer , Integer>();
         for ( int i = 0 ; i < MATRIX_SIZE * MATRIX_SIZE ; i++) {
             if ( board[i / MATRIX_SIZE][i % MATRIX_SIZE ] == '.' ) {
                miss [missI++] = i;
             } else
                setFlag (i, false);
         }
        missL = missI;
         for ( missI = 0; true;) {
             if ( missI < 0)
                 return;
             int i = miss[missI ];
             if ( board[i / MATRIX_SIZE][i % MATRIX_SIZE ] < '1' )
                board [i / MATRIX_SIZE][i % MATRIX_SIZE] = '1';
             if ( board[i / MATRIX_SIZE][i % MATRIX_SIZE ] > '9' ) {
                board [i / MATRIX_SIZE][i % MATRIX_SIZE] = '.';
                 if(setRecord .containsKey(i)){
                    removeFlag (i, setRecord.get (i));
                    setRecord .remove( i);
                 }
                 if (-- missI < 0)
                     return;
                i = miss[ missI];
                 if(setRecord .containsKey(i)){
                    removeFlag (i, setRecord.get (i));
                    setRecord .remove( i);
                 }
                board [i / MATRIX_SIZE][i % MATRIX_SIZE]++;
                 continue;
             }
             if ( check(i )) {
                 if ( missI == missL - 1 )
                     return;
                setFlag (i, true);
                missI ++;
             } else {
                board [i / MATRIX_SIZE][i % MATRIX_SIZE]++;
             }
         }
    }

    public boolean check (int index ) {
         if ( board[index / MATRIX_SIZE][index % MATRIX_SIZE ] == '.' )
             return false ;
         int mask = 1 << (board [index / MATRIX_SIZE][index % MATRIX_SIZE] - '0' );
         if (( rows[index / MATRIX_SIZE] & mask ) != 0
                 || ( cols[index % MATRIX_SIZE] & mask ) != 0
                 || ( cubes[index / MATRIX_SIZE / 3][ index % MATRIX_SIZE / 3] & mask) != 0)
             return false ;
         return true ;
    }

    public void removeFlag (int index , int value) {
         if ( board[index / MATRIX_SIZE][index % MATRIX_SIZE ] == '.' )
             return;
         int mask = ~(1 << (board[index / MATRIX_SIZE][index % MATRIX_SIZE] - '0' ));
        rows [index / MATRIX_SIZE] &= mask;
        cols [index % MATRIX_SIZE] &= mask;
        cubes [index / MATRIX_SIZE / 3][index % MATRIX_SIZE / 3] &= mask;
    }

    public void setFlag (int index , boolean ifRecord) {
         if ( board[index / MATRIX_SIZE][index % MATRIX_SIZE ] == '.' )
             return;
         int mask = 1 << (board [index / MATRIX_SIZE][index % MATRIX_SIZE] - '0' );
        rows [index / MATRIX_SIZE] |= mask;
        cols [index % MATRIX_SIZE] |= mask;
        cubes [index / MATRIX_SIZE / 3][index % MATRIX_SIZE / 3] |= mask;
         if ( ifRecord) {
            setRecord .put( index,
                    board [index / MATRIX_SIZE][index % MATRIX_SIZE] - '0');
         }
    }
}
