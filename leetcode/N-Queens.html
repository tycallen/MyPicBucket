N-Queens

创建时间：2014/6/20 18:27
更新时间：2014/6/20 18:27

The n-queens puzzle is the problem of placing n queens on an n×n chessboard such that no two queens attack each other.

Given an integer n, return all distinct solutions to the n-queens puzzle.

Each solution contains a distinct board configuration of the n-queens' placement, where 'Q' and '.' both indicate a queen and an empty space respectively.

For example,
There exist two distinct solutions to the 4-queens puzzle:

[
 [".Q..",  // Solution 1
  "...Q",
  "Q...",
  "..Q."],

 ["..Q.",  // Solution 2
  "Q...",
  "...Q",
  ".Q.."]
]

public class Solution {
    List<String []> result;

    public List <String[]> solveNQueens(int n) {
        result = new ArrayList<String []>();
         if(n == 1 ){
            result .add( new String[]{"Q" });
             return result;
         }
         if ( n <= 3)
             return result;
         int[] rowPosi = new int [n];
         int row = 0 ;
         while ( rowPosi[0 ] < n) {
             if ( rowPosi[row ] >= n) {
                rowPosi [row] = 0;
                rowPosi [--row]++;
             } else {
                 if ( check(rowPosi , row)) {
                     if ( row < n - 1 ) {
                        row ++;
                     } else {
                        outPut (rowPosi);
                        rowPosi [row]++;
                     }
                 } else {
                    rowPosi [row]++;
                 }
             }
         }
         return result;
    }

    public void outPut (int[] rowPosi) {
        String [] outPut = new String [rowPosi.length];
         for ( int i = 0 ; i < rowPosi.length ; i++) {
            StringBuilder row = new StringBuilder();
             for ( int j = 0 ; j < rowPosi.length ; j++) {
                 if ( j == rowPosi[i ])
                    row .append( "Q");
                 else
                    row .append( ".");
             }
            outPut [i] = row.toString ();
         }
        result .add( outPut);
    }

    public boolean check (int[] rowPosi, int row) {
         for ( int i = 0 ; i < row; i ++) {
             if ( rowPosi[i ] == rowPosi[row ]
                     || rowPosi[row] == rowPosi[i] + row - i
                     || rowPosi[row] == rowPosi[i] - row + i)
                 return false ;
         }
         return true ;
    }
}
