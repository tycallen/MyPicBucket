Given two binary strings, return their sum (also a binary string).

For example,
a = "11"
b = "1"
Return "100".

public class Solution {
    public String addBinary (String a, String b) {
         if ( a == null || a. length() == 0 )
             return b;
         if ( b == null || b. length() == 0 )
             return a;
         int longer = Math.max (a. length(), b .length());
         int shorter = Math.min (a. length(), b .length());
         char[] added = new char [longer + 1];
         int index = 0;
         boolean carry = false;
         while ( index < shorter ) {
             if ( a.charAt(a .length() - 1 - index ) == '1'
                     && b. charAt(b .length() - 1 - index ) == '1' ) {
                 if ( carry)
                    added [longer - index] = '1' ;
                 else
                    added [longer - index] = '0' ;
                carry = true ;
             } else if (a .charAt( a.length() - 1 - index) == '1'
                     ^ b. charAt(b .length() - 1 - index ) == '1' ) {
                 if ( carry)
                    added [longer - index] = '0' ;
                 else
                    added [longer - index] = '1' ;
             } else {
                 if ( carry)
                    added [longer - index] = '1' ;
                 else
                    added [longer - index] = '0' ;
                carry = false ;
             }
            index ++;
         }
        String temp = a. length() > b. length() ? a : b;
         while ( index < longer ) {
             if ( temp.charAt (longer - 1 - index ) == '1' && carry) {
                added [longer - index] = '0' ;
             } else if (temp .charAt( longer - 1 - index) == '1' ^ carry ) {
                added [longer - index] = '1' ;
                carry = false ;
             } else
                added [longer - index] = '0' ;
            index ++;
         }
         if ( carry)
            added [0] = '1';
         return carry ? String.copyValueOf (added) : String.copyValueOf(added, 1,
                longer );
    }
}

public String addBinary(String a, String b ) {
    int m = a .length();
    int n = b .length();
    int carry = 0 ;
    String res = "";
    // the final length of the result depends on the bigger length between a and b,
    // (also the value of carry, if carry = 1, add "1" at the head of result, otherwise)
    int maxLen = Math. max(m , n);
    for (int i = 0; i < maxLen; i++) {
        // start from last char of a and b
        // notice that left side is int and right side is char
        // so we need to  minus the decimal value of '0'
        int p = i < m ? a. charAt(m - 1 - i) - '0' : 0;
        int q = i < n ? b. charAt(n - 1 - i) - '0' : 0;
        int tmp = p + q + carry;
        carry = tmp / 2 ;
        res = tmp % 2 + res;
    }
    return (carry == 0 ) ? res : "1" + res;
}
