Symmetric Tree

创建时间：2014/6/9 16:33
更新时间：2014/6/9 16:38

Given a binary tree, check whether it is a mirror of itself (ie, symmetric around its center).

For example, this binary tree is symmetric:

    1
   / \
  2   2
 / \ / \
3  4 4  3

But the following is not:

    1
   / \
  2   2
   \   \
   3    3

Note:
Bonus points if you could solve it both recursively and iteratively.

confused what "{1,#,2,3}" means? > read more on how binary tree is serialized on OJ.

/**
 * Definition for binary tree
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
 import java .util. Stack;
public class Solution {
    public boolean isSymmetric (TreeNode root) {
         if ( null == root)
             return true ;
        Stack <TreeNode> stack = new Stack< TreeNode>();
        stack .push( root);
         while (! stack.empty ()) {
             // get current layer
             int size = stack.size ();
            TreeNode [] currLayer = new TreeNode [size];
             int i = 0 ;
             while (! stack.empty ()) {
                currLayer [i++] = stack.pop ();
             }
             boolean isThisLayerEmpty = true;
             if(size == 1 ){
                 if(currLayer [0]!= null)
                    isThisLayerEmpty = false ;
             }
             // check current layer
             for ( i = 0; i < size / 2 ; i++) {
                 if ( currLayer[i ]== null && currLayer[size - i - 1]==null)
                     continue;
                 else if (currLayer[i]== null ^ currLayer[size - i - 1]==null){
                     return false ;
                 }else{
                    isThisLayerEmpty = false ;
                     if(currLayer [i]. val != currLayer [size - i - 1].val )
                         return false ;
                 }
             }
             if(isThisLayerEmpty ) return true;
             for ( i = 0; i < size ; i++) {
                if(null == currLayer [i]) continue;
                stack .push( currLayer[i ].left);
                stack .push( currLayer[i ].right);
             }
         }
         return true ;
    }
}

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool check(TreeNode *leftNode, TreeNode *rightNode )
    {
        if ( leftNode == NULL && rightNode == NULL)
            return true ;
           
        if ( leftNode == NULL || rightNode == NULL)
            return false ;
           
        return leftNode->val == rightNode ->val && check(leftNode ->left, rightNode->right) &&
            check (leftNode->right, rightNode ->left);
    }
   
    bool isSymmetric (TreeNode *root) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if ( root == NULL )
            return true ;
           
        return check( root->left , root-> right);
    }
};
