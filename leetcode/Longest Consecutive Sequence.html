Longest Consecutive Sequence

创建时间：2014/6/17 16:37
更新时间：2014/6/17 16:38

Given an unsorted array of integers, find the length of the longest consecutive elements sequence.

For example,
Given [100, 4, 200, 1, 3, 2],
The longest consecutive elements sequence is [1, 2, 3, 4]. Return its length: 4.

Your algorithm should run in O(n) complexity.

public class Solution {
    public int longestConsecutive (int[] num) {
         int max = 1 ;
         if ( num == null || num.length == 0 )
             return max;
        Map <Integer, Integer> head = new HashMap<Integer , Integer>();
        Map <Integer, Integer> tail = new HashMap<Integer , Integer>();
         boolean isHead, isTail;
         int newH, newT;
         for ( int i = 0 ; i < num.length ; i++) {
            isHead = head. containsKey(num [i] + 1);
            isTail = tail. containsKey(num [i] - 1);
             if (! head.containsKey (num[ i]) && !tail. containsKey(num [i])) {
                 if ( isHead && isTail ) {
                    newH = tail. remove(num [i] - 1);
                    newT = head. remove(num [i] + 1);
                    tail .put( newT, newH );
                    head .put( newH, newT );
                    max = Math. max(max , newT - newH + 1);
                 } else if (isHead ) {
                    newT = head. remove(num [i] + 1);
                    head .put( num[i ], newT);
                    tail .put( newT, num [i]);
                    max = Math. max(max , newT - num[i ] + 1 );
                 } else if (isTail ) {
                    newH = tail. remove(num [i] - 1);
                    tail .put( num[i ], newH);
                    head .put( newH, num [i]);
                    max = Math. max(max , num[ i] - newH + 1 );
                 } else {
                    head .put( num[i ], num[ i]);
                    tail .put( num[i ], num[ i]);
                 }
             }
         }
         return max;
    }
}
