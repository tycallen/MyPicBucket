Partition List

创建时间：2014/6/19 12:06
更新时间：2014/6/19 13:45

Given a linked list and a value x, partition it such that all nodes less than x come before nodes greater than or equal to x.

You should preserve the original relative order of the nodes in each of the two partitions.

For example,
Given 1->4->3->2->5->2 and x = 3,
return 1->2->2->4->3->5.

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode partition (ListNode head, int x) {
        ListNode headPointer = new ListNode(Integer .MIN_VALUE);
        headPointer .next = head;
        ListNode lastSmaller , firstBigger = null, lastBigger = null;
        lastSmaller = headPointer;
         while ( lastSmaller.next != null ) {
             while ( lastSmaller.next != null && lastSmaller.next .val < x)
                lastSmaller = lastSmaller.next;
             if ( firstBigger == null) {
                firstBigger = lastSmaller.next;
                lastBigger = firstBigger;
             } else {
                lastBigger .next = lastSmaller.next ;
                lastBigger = lastBigger.next;
             }
             if(lastBigger == null )
                 break;
             while ( lastBigger.next != null && lastBigger.next .val >= x)
                lastBigger = lastBigger.next;
            lastSmaller .next = lastBigger.next ;
             if(lastBigger .next != null){
                lastSmaller = lastSmaller.next;
             }
         }
        lastSmaller .next = firstBigger;
         if(lastBigger != null )
            lastBigger .next = null;
         return headPointer.next;
    }
}
