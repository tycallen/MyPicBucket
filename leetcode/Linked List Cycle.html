Linked List Cycle

创建时间：2014/6/7 20:56
更新时间：2014/6/7 20:57

Given a linked list, determine if it has a cycle in it.

Follow up:
Can you solve it without using extra space?

/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public boolean hasCycle (ListNode head) {
        ListNode faster ,slower;
        
        faster = head;
        slower = head;
       
         while ( faster != null && slower != null) {
            slower = slower.next;
            faster = faster.next;
             if(null == faster ) return false;
            faster = faster.next;
             if(slower .equals( faster)) return true ;
         }
       
         return false ;
    }
    
}
