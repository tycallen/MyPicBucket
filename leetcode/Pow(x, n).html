Implement pow(x, n).

public class Solution {
         double epsilon = 0.00000001;

    public double pow (double x, int n) {
         if ( n == 0)
             return 1 ;
         if ( x < epsilon && x > - epsilon) {
             if ( n > 0)
                 return 1 ;
             else if (x > 0)
                 return Double.MAX_VALUE;
             else
                 return - Double.MAX_VALUE ;
         }
         if(x < 1 + epsilon && x > 1 - epsilon)
             return 1 ;
         if(x < - 1 + epsilon && x > -1 - epsilon )
            return n% 2==0 ?1:- 1;
         if ( n == Integer.MAX_VALUE ) {
             if ( x > 1)
                 return Integer.MAX_VALUE;
             else if (x < -1)
                 return - Integer.MAX_VALUE ;
             else
                 return 0 ;
         } else if (n == - Integer.MAX_VALUE ) {
             if ( x < -1 || x > 1)
                 return 0 ;
             else if (x > -1 && x < 0 )
                 return - Integer.MAX_VALUE ;
             else
                 return Integer.MAX_VALUE;
         }
        
         double result = x;
         int abs = Math.abs (n);
         int count = 1;
         while ( count * 2 < abs) {
            result *= result;
            count *= 2 ;
         }
         return n > 0 ? result * pow (x, abs - count ) : 1 / result
                 / pow( x, abs - count );
    }
}
