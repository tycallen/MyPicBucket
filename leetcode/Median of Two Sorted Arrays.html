public class Solution {
    public double findMedianSortedArrays (int A[], int B[]) {
         int i = 0 ;
         if ( A == null && B == null)
             return 0 ;
         if ( A != null)
            i += A. length;
         if ( B != null)
            i += B. length;
         if (( i & 1) != 0)
             return findKNumSortedArrays(A, B , i / 2 + 1);
         return ( double) (findKNumSortedArrays(A, B , i / 2 + 1) + findKNumSortedArrays(
                A , B, i / 2)) / 2 ;
    }

    public int findKNumSortedArrays (int A[], int as, int ae, int B[], int bs,
             int be, int k) {
         if ( A == null || B == null || k > ae + be - as - bs || k <= 0)
             return 0 ;
         int la = ae - as > 0 ? ae - as : 0 ;
         int lb = be - bs > 0 ? be - bs : 0 ;
         int i = (int) ((la + 0.0 ) / (la + lb ) * (k - 1) + as);
         int j = (k - 1) - i + as + bs ;

         int Ai = (i >= ae ? Integer.MAX_VALUE : A [i]);
         int Ai_1 = (i - 1 < as ? Integer.MIN_VALUE : A[ i - 1]);
         int Bj = (j >= be ? Integer.MAX_VALUE : B [j]);
         int Bj_1 = (j - 1 < bs ? Integer.MIN_VALUE : B[ j - 1]);

         if ( Ai <= Bj && Ai >= Bj_1)
             return Ai;
         if ( Bj <= Ai && Bj >= Ai_1)
             return Bj;

         assert (( Ai <= Bj && Ai <= Bj_1) || ( Bj <= Ai && Bj <= Ai_1));

         if ( Ai < Bj )
             return findKNumSortedArrays(A, i + 1 , ae, B, bs, j , k - i - 1 + as);
         else
             return findKNumSortedArrays(A, as , i, B, j + 1, be , k - j - 1 + bs);
    }

    public int findKNumSortedArrays (int[] A, int[] B, int k) {
         return findKNumSortedArrays(A, 0, A .length, B, 0, B.length, k );
    }
}
