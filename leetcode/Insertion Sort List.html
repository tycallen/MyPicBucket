Sort a linked list using insertion sort.

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode insertionSortList (ListNode head) {
         if ( head == null)
             return head;
        ListNode pointer = new ListNode(0 );
        pointer .next = head;
        ListNode p = head. next, insert , temp1, temp2;
        head .next = null;
         while ( p != null) {
             for ( insert = pointer ; insert.next != null
                     && insert.next.val < p. val; insert = insert.next)
                 ;
            temp1 = insert.next;
            insert .next = p;
            temp2 = p. next;
            p .next = temp1;
            p = temp2;
         }
         return pointer.next;
    }
}
