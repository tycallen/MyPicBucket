Given an array where elements are sorted in ascending order, convert it to a height balanced BST.

import java. util.List ;

public class Solution {
    public TreeNode sortedArrayToBST (int[] num) {
         if ( num == null || num.length == 0 )
             return null ;
        TreeNode root = new TreeNode(num [0]);
         for ( int i = 1 ; i < num.length ; i++) {
            insertBst (num[ i], root);
         }
         return root;
    }

    public void insertBst (int val, TreeNode root ) {
         if ( root == null) {
            root = new TreeNode(val );
             return;
         }
         if ( val < root .val) {
             if ( root.left != null )
                insertBst (val, root.left );
             else {
                root .left = new TreeNode(val );
             }
             if ( Math.abs (maxDepth(root.left ) - maxDepth(root .right)) == 2) {
                 if ( val < root .left. val) {
                     // LL型
                    rotateLL (root);
                 } else {
                     // LR型
                    rotateLR (root);
                 }
             }
         } else {
             if ( root.right != null ){
                insertBst (val, root.right );
             }
             else {
                root .right = new TreeNode(val );
             }
             if ( Math.abs (maxDepth(root.left ) - maxDepth(root .right)) == 2) {
                 if ( val < root .right. val) {
                     // RL型
                    rotateRL (root);
                 } else {
                     // RR型
                    rotateRR (root);
                 }
             }
         }
    }
    
    public void rotateLL (TreeNode root){
        TreeNode temp = root. left.right ;
         int valTemp = root.val ;
        root .val = root.left .val;
        root .left. val = valTemp ;
        root .left. right = root .right;
        root .right = root.left ;
        root .left = root.left .left;
        root .right. left = temp ;
    }
    
    public void rotateRR (TreeNode root){
        TreeNode temp = root. right.left ;
         int valTemp = root.val ;
        root .val = root.right .val;
        root .right. val = valTemp ;
        root .right. left = root .left;
        root .left = root.right ;
        root .right = root.right .right;
        root .left. right = temp ;
        
    }
    
    public void rotateLR (TreeNode root){
        rotateRR (root. left);
        rotateLL (root);
    }
    
    public void rotateRL (TreeNode root){
        rotateLL (root. right);
        rotateRR (root);
    }

    public int maxDepth (TreeNode root) {
         if ( root == null)
             return 0 ;
         int l, r;
        l = maxDepth(root.left );
        r = maxDepth(root.right );
         return ( l > r ? l : r) + 1;
    }

这道题是已排序的，故不需要旋转操作
/**
 * Definition for binary tree
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
   public TreeNode sortedArrayToBST (int[] num) {
         if ( num == null || num.length == 0 )
             return null ;
        
         return sortedArrayToBST(num,0 ,num. length-1 );
    }

    public TreeNode sortedArrayToBST (int[] num,int s,int e){
         if ( num == null || num.length == 0 || s > e )
             return null ;
         if(s == e) return new TreeNode (num[ s]);
         int mid = (e-s )/2+ s;
        TreeNode root = new TreeNode(num [mid]);
        root .left = sortedArrayToBST(num ,s, mid-1 );
        root .right = sortedArrayToBST(num ,mid+ 1,e );
         return root;
    }
}
